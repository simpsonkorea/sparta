function shareProduct(title, price, url) { if (Kakao.isInitialized()) { Kakao.Link.sendDefault({ objectType: "feed", content: { title: title, description: `${title}: ${price}에 구매하기!`, imageUrl: "https://example.com/path/to/your/image.jpg", // Add a representative image for the shared content link: { webUrl: url, mobileWebUrl: url, }, }, buttons: [ { title: "자세히 보기", link: { webUrl: url, mobileWebUrl: url, }, }, ], }); } else { alert("카카오 SDK를 초기화하지 못했습니다."); } }
